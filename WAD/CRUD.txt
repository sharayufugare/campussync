server.js=>
const express = require('express');

const mongoose = require('mongoose');

const bodyParser = require('body-parser');

const path = require('path');

const app = express();

const PORT = 3000;

// Middleware
app.use(bodyParser.json());

// MongoDB Connection
mongoose.connect('mongodb://127.0.0.1:27017/crudapp')
.then(() => console.log('MongoDB Connected'))
.catch(err => console.log(err));

// Schema
const userSchema = new mongoose.Schema({

  name: String,
  email: String

});

// Model
const User = mongoose.model('User', userSchema);


// CREATE USER API
app.post('/users', async (req, res) => {

  const { name, email } = req.body;

  const existingUser = await User.findOne({

    $or: [{ name }, { email }]

  });

  if (existingUser) {

    return res.status(400).json({

      message: 'User already exists'

    });

  }

  const user = new User({

    name,
    email

  });

  await user.save();

  res.status(201).json(user);

});


// GET USERS API
app.get('/users', async (req, res) => {

  const users = await User.find();

  res.json(users);

});


// GET USER BY NAME
app.get('/users/name/:name', async (req, res) => {

  const user = await User.findOne({

    name: req.params.name

  });

  if (!user) {

    return res.status(404).json({

      message: 'User not found'

    });

  }

  res.json(user);

});


// UPDATE USER
app.put('/users/name/:name', async (req, res) => {

  const user = await User.findOne({

    name: req.params.name

  });

  if (!user) {

    return res.status(404).json({

      message: 'User not found'

    });

  }

  user.name = req.body.newName;

  user.email = req.body.email;

  await user.save();

  res.json(user);

});


// DELETE USER
app.delete('/users/name/:name', async (req, res) => {

  const user = await User.findOne({

    name: req.params.name

  });

  if (!user) {

    return res.status(404).json({

      message: 'User not found'

    });

  }

  await User.deleteOne({

    name: req.params.name

  });

  res.json({

    message: 'User deleted successfully'

  });

});


// Serve HTML
app.get('/', (req, res) => {

  res.sendFile(path.join(__dirname, 'public', 'index.html'));

});


// Start Server
app.listen(PORT, () => {

  console.log(`Server running at http://localhost:${PORT}`);

});

const API = 'http://localhost:3000/users';


// CREATE
function createUser() {

  const name = document.getElementById('name').value;

  const email = document.getElementById('email').value;

  fetch(API, {

    method: 'POST',

    headers: {

      'Content-Type': 'application/json'

    },

    body: JSON.stringify({

      name,
      email

    })

  })

  .then(res => res.json())

  .then(data => {

    alert(data.message || 'User Created');

    getUsers();

  });

}


// READ
function getUsers() {

  fetch(API)

  .then(res => res.json())

  .then(users => {

    const list = document.getElementById('userList');

    list.innerHTML = '';

    users.forEach(user => {

      const li = document.createElement('li');

      li.innerText = `${user.name} - ${user.email}`;

      list.appendChild(li);

    });

  });

}


// UPDATE
function updateUser() {

  const oldName = document.getElementById('oldName').value;

  const newName = document.getElementById('newName').value;

  const newEmail = document.getElementById('newEmail').value;

  fetch(`${API}/name/${oldName}`, {

    method: 'PUT',

    headers: {

      'Content-Type': 'application/json'

    },

    body: JSON.stringify({

      newName,
      email: newEmail

    })

  })

  .then(res => res.json())

  .then(data => {

    alert(data.message || 'User Updated');

    getUsers();

  });

}


// DELETE
function deleteUser() {

  const deleteName = document.getElementById('deleteName').value;

  fetch(`${API}/name/${deleteName}`, {

    method: 'DELETE'

  })

  .then(res => res.json())

  .then(data => {

    alert(data.message);

    getUsers();

  });

}

</script>

</body>
</html>

what is async , await?
async and await are used in JavaScript to handle asynchronous operations in a simple way

Make function asynchronous” means:
“Don’t block the program while doing slow work. Let it continue other tasks and handle result later.”

1. Role of Mongoose in Node.js

Mongoose is used to connect Node.js with MongoDB.
It helps to define schema, validate data, and perform database operations easily.

2. Purpose of body-parser middleware

Express.js middleware body-parser is used to read JSON data from request body and convert it into JavaScript object so server can use it.

3. How POST method works in CRUD API

POST method is used to create new data in the database.
It takes data from request body (req.body) and saves it into MongoDB using Mongoose.

4. Difference between PUT and GET
GET	PUT
Used to read data	Used to update data
Does not change data	Modifies existing data
Data sent in URL	Data sent in request body
5. findOne() vs find()
findOne() → returns only one matching document
find() → returns all matching documents (array)
6. deleteOne() vs deleteMany()
deleteOne() → deletes only first matching record
deleteMany() → deletes all matching records
