****************app.js**************


document.getElementById('registrationForm').addEventListener('submit', function(e) {

  e.preventDefault(); //Stops page refresh after submit

  // Get form values
  const userData = {
    name: document.getElementById('name').value,
    email: document.getElementById('email').value,
    password: document.getElementById('password').value
  };

  // AJAX Request
  let xhr = new XMLHttpRequest(); //Creates request object

  xhr.open("POST", "https://jsonplaceholder.typicode.com/posts", true); //POST → send data ,fake API (testing server) , async = true

  xhr.setRequestHeader("Content-Type", "application/json"); //Data is sent in JSON format

  xhr.onload = function () { //Runs when server responds

    if (xhr.status === 201 || xhr.status === 200) { //Means request successful

      // Get existing users
      let users = JSON.parse(localStorage.getItem('users')) || []; //Get existing users , If none → empty array

      // Check duplicate email
      const userExists = users.some(user => user.email === userData.email);

      if (!userExists) {

        // Add new user
        users.push(userData);

        // Save to localStorage
        localStorage.setItem('users', JSON.stringify(users)); //Stores data permanently in browser

        alert("Registration Successful");

        // Redirect
        window.location.href = "user-list.html";

      } else {
        alert("User already exists");
      }

    } else {
      alert("Error saving data");
    }

  };

  // Send data
  xhr.send(JSON.stringify(userData)); //is used in AJAX (XMLHttpRequest) to send data to a server.

});

*******************user-list.js***************

const users = JSON.parse(localStorage.getItem('users')) || []; //converts string → JavaScript object/array

const userList = document.getElementById('userList'); 

// Display users
if (users.length === 0) {

  userList.innerHTML = "<li>No users registered</li>";

} else {

  users.forEach(user => {

    const li = document.createElement('li');

    li.textContent = `Name: ${user.name}, Email: ${user.email}`;

    userList.appendChild(li);//Adds list item into HTML page

  });

}

// Clear localStorage
document.getElementById('clearDataBtn').addEventListener('click', function() { //When user clicks “Clear Data” button

  localStorage.removeItem('users'); //Deletes all stored users

  alert("Data Cleared");

  window.location.reload();

});
