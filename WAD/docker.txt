Check Docker version
docker --version

# Pull Ubuntu image
docker pull ubuntu

# Check images
docker images

# Run Ubuntu container
docker run -it ubuntu /bin/bash

# Inside container
apt update
apt install apache2 -y

# Exit container
exit

# Check containers
docker ps -a

# Commit container into new image
docker commit -a "Shaily" -m "Installed Apache" container_id ubuntu-apache

# Check images again
docker images

# Run new image
docker run -it ubuntu-apache /bin/bash

# Exit container
Exit
