server.js=>
const express = require('express'); //Loads Express module Used to create web server

const app = express(); //Creates Express application

const path = require('path');

// Serve static files - Serve all files inside public folder directly”
app.use(express.static(path.join(__dirname, 'public')));

// Homepage route - When user opens:http://localhost:3000/ This function runs.
app.get('/', (req, res) => {

  res.sendFile(path.join(__dirname, 'public', 'index.html')); //Send index.html file to browser

});

// Server Port
const PORT = 3000;

// Start server
app.listen(PORT, () => {

  console.log(`Server running at http://localhost:${PORT}`);

});
